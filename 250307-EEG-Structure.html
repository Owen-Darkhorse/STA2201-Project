multiLinePlot <- function(mat) {
  n <- nrow(mat)
  p <- ncol(mat)
  mat <- cbind("Index" = 1:n, mat) %>% as.data.frame()
  
  g <- ggplot(data = mat)
  for (j in 2:(p+1)) {
    g <- g + geom_line(aes_string(x = "Index", 
                            y = paste0("V", j),
                            color = factor(j)))
  }
  g + 
    ggtitle("Time Series of EEG")+
    scale_color_discrete()
  
  print(g)
}
